  class AboutMe {
    private age: number;
    private likes: string;

    constructor(params:{age: number, likes: string}){
      const {age, likes } =  params;
      this.age = age;
      this.likes = likes;
    }
    get info(): string {
      return `I'm ${this.age} Software engineering student and Backend developer from venezuela ${this.likes}`;
    }
}
